1) Add loader (single source of truth)

File: src/lib/mapsLoader.ts

import { Loader } from '@googlemaps/js-api-loader';

let loader: Loader | null = null;
let loaded = false;

export async function loadMapsOnce() {
  if (loaded && (window as any).google?.maps?.places) return (window as any).google;

  const key = import.meta.env.VITE_GOOGLE_MAPS_API_KEY;
  if (!key) throw new Error('Missing VITE_GOOGLE_MAPS_API_KEY');

  if (!loader) loader = new Loader({ apiKey: key, libraries: ['places'] });

  await loader.load();
  if (!(window as any).google?.maps?.places) {
    throw new Error('Places library failed to load');
  }
  loaded = true;
  return (window as any).google;
}

2) Known-good Autocomplete component (no freeze)

File: src/components/LocationInput.tsx

import { useEffect, useRef } from 'react';
import { loadMapsOnce } from '@/lib/mapsLoader';

export type Address = {
  street: string; city: string; state: string; postalCode: string;
  country: string; place_id: string; formatted_address: string;
};

export default function LocationInput({
  value, onChange, onAddressSelected, placeholder = 'Enter address', disabled = false,
}: {
  value: string; onChange: (v: string) => void;
  onAddressSelected: (a: Address) => void;
  placeholder?: string; disabled?: boolean;
}) {
  const inputRef = useRef<HTMLInputElement | null>(null);

  useEffect(() => {
    let active = true;
    loadMapsOnce().then(() => {
      if (!active || !inputRef.current) return;

      const ac = new google.maps.places.Autocomplete(inputRef.current!, {
        fields: ['address_components', 'formatted_address', 'place_id'],
        types: ['address'],
      });

      ac.addListener('place_changed', () => {
        const p = ac.getPlace();
        if (!p?.address_components) return;
        const get = (t: string) => p.address_components!.find(c => c.types.includes(t))?.long_name || '';
        const street = [get('street_number'), get('route')].filter(Boolean).join(' ').trim();
        const city = get('locality') || get('sublocality') || get('postal_town');
        const state = get('administrative_area_level_1');
        const postalCode = get('postal_code');
        const country = get('country');

        onAddressSelected({
          street, city, state, postalCode, country,
          place_id: p.place_id || '',
          formatted_address: p.formatted_address || '',
        });
      });
    });

    return () => { active = false; };
  }, []);

  return (
    <input
      ref={inputRef}
      value={value}
      onChange={e => onChange(e.target.value)}   // never toggle readOnly/disabled here
      placeholder={placeholder}
      disabled={disabled}
      autoComplete="off"
      inputMode="text"
    />
  );
}

<!doctype html><meta charset="utf-8"><title>Places Check</title>
<input id="addr" placeholder="Type an address..." style="width:360px;padding:8px;font-size:16px">
<pre id="out"></pre>
<script>
  (async () => {
    const key = (window.location.search.match(/[?&]key=([^&]+)/)||[])[1] || '';
    const s = document.createElement('script');
    s.src = `https://maps.googleapis.com/maps/api/js?key=${key}&libraries=places`;
    s.onload = () => {
      const ac = new google.maps.places.Autocomplete(document.getElementById('addr'), {
        fields: ['address_components','formatted_address','place_id'], types:['address']
      });
      ac.addListener('place_changed', () => {
        const p = ac.getPlace();
        document.getElementById('out').textContent =
          JSON.stringify({ ok:true, place_id:p.place_id, addr:p.formatted_address }, null, 2);
      });
    };
    document.body.appendChild(s);
  })();
</script>
