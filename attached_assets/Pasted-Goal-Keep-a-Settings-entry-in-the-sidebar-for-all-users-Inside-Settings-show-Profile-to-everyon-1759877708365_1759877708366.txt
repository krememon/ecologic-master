1) Build a role-aware Settings nav
// SettingsLayout.tsx
import { useCan } from "@/hooks/useCan";

const SettingsLayout = ({ children }) => {
  const { user } = useSession(); // however you fetch session
  const canViewCompany = useCan("org.view"); // Owner/Supervisor only

  const links = [
    { label: "Profile", href: "/settings/profile", key: "profile", always: true },
    canViewCompany && { label: "Company", href: "/settings/company", key: "company" },
  ].filter(Boolean);

  // If user lands on /settings, redirect to first accessible tab
  useEffect(() => {
    if (router.pathname === "/settings") {
      router.replace(links[0].href);
    }
  }, [router.pathname, links]);

  return (
    <div className="grid gap-6 md:grid-cols-[220px_1fr]">
      <aside className="rounded-xl border p-2">
        <nav className="flex md:flex-col gap-2">
          {links.map((l) => (
            <Link key={l.key} href={l.href} className={linkStyles(l.href)}>
              {l.label}
            </Link>
          ))}
        </nav>
      </aside>
      <main>{children}</main>
    </div>
  );
};
export default SettingsLayout;

2) Guard the Company page (render nothing if unauthorized)
// /settings/company.tsx
import { useCan } from "@/hooks/useCan";
export default function CompanySettings() {
  const canViewCompany = useCan("org.view");
  if (!canViewCompany) return null; // no 403; completely hidden

  // Owner-only controls:
  const canManageOrg = useCan("org.manage");

  return (
    <CompanyCard
      canEdit={canManageOrg}
      canRotateCode={canManageOrg}
      // shows Invite Code readonly field and Copy button
    />
  );
}
