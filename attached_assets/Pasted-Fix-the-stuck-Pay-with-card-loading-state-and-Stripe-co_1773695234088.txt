      job?.clientEmail ||
      user?.email ||

    console.log('[stripe-pay] invoiceId=%s', invoice?.id);
    console.log('[stripe-pay] billingEmail=%s', billingEmail);
    console.log('[stripe-pay] submitting=true');

    const result = await stripe.confirmPayment({

      confirmParams: {

          billing_details: {
            email: billingEmail,
          },
        },
        return_url: returnUrl,
      },
      redirect: 'if_required',
    });

    if (result.error) {
      throw result.error;
    }

    // success handling here
  } catch (err: any) {
    console.error('[stripe-pay] confirmPayment error=', err);
    setError(err?.message || 'Payment failed');
  } finally {
    console.log('[stripe-pay] submitting=false');
    setSubmitting(false);
  }
